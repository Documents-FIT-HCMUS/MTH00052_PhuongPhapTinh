function [yc] = XapXiDaThucLagrange(xx,yy,xc,isPlot)
syms x;
n=length(xx); L=0;
for i=1:n
    Tu = 1;
    Mau = 1;
    for j=1:n
        if i~=j
            Tu = Tu*(x-xx(j));
            Mau = Mau*(xx(i)-xx(j));
            
        end
    end
    L = L + (Tu/Mau)*yy(i);
end

yc = subs(L,x,xc);
yc = double(yc);
if isPlot==1
    fplot(L,[xx(1) xx(end)]);
    hold on ;
    plot(xx,yy,'bo');
end
end

function[] = xapxidathucNewton(xx, yy, xc)
    syms x
    n = length(xx);
    n1 = yy;
    tsp = zeros(1, n);
    N = n1(1);
    for i = 2:n
       for j = 1:(n-i+1)
          tsp(j)=(n1(j+1)-n1(j))/(xx(j+i-1)-xx(j));
       end
       tich = 1;
       for k = 1:(i-1)
          tich = tich*(x-xx(k));
       end
       N = N + tsp(1)*tich;
       n1 = tsp;
    end
    yc = subs(N, x, xc);
    ezplot(N, [xx(1) xx(end)]);
    hold on;
    plot(xx, yy, 'ro');
    title = {'xx', 'yy'};
    data = zeros(n+length(xc), length(title));
    for i = 1:n
       data(i,:)= [xx(i) yy(i)]; 
    end
    for j = n+1:n+length(xc)
       data(j,:)=[xc(j-n) yc(j-n)]; 
    end
    dt = figure();
    t = uitable(dt, 'data', data, 'Columnname', title, 'Position', [0 0 400 400]);
    disp(t);
end
